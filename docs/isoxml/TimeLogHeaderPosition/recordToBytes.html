Uint8List recordToBytes(Position position) {
  final byteData = ByteData(byteLength);
  var currentOffset = 0;
  if (readNorth) {
    byteData.setInt32(
      currentOffset,
      (position.north * 1e7).round(),
      Endian.little,
    );
    currentOffset += 4;
  }
  if (readEast) {
    byteData.setInt32(
      currentOffset,
      (position.east * 1e7).round(),
      Endian.little,
    );
    currentOffset += 4;
  }
  if (readUp) {
    byteData.setInt32(currentOffset, position.up ?? 0, Endian.little);
    currentOffset += 4;
  }
  if (readStatus) {
    byteData.setUint8(currentOffset, position.status.value);
    currentOffset++;
  }
  if (readPdop) {
    byteData.setUint16(
      currentOffset,
      ((position.pdop ?? 0) * 1e1).round(),
      Endian.little,
    );
    currentOffset += 2;
  }
  if (readHdop) {
    byteData.setUint16(
      currentOffset,
      ((position.hdop ?? 0) * 1e1).round(),
      Endian.little,
    );
    currentOffset += 2;
  }
  if (readNumberOfSatellites) {
    byteData.setUint8(currentOffset, position.numberOfSatellites ?? 0);
    currentOffset++;
  }
  if (readgpsUtcTimeMs) {
    byteData.setUint32(
      currentOffset,
      position.gpsUtcTimeMs ?? 0,
      Endian.little,
    );
    currentOffset += 4;
  }
  if (readGpsUtcDate) {
    byteData.setUint16(
      currentOffset,
      position.gpsUtcDate ?? 0,
      Endian.little,
    );
    currentOffset += 2;
  }

  return byteData.buffer.asUint8List(0, currentOffset);
}
